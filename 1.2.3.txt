(3. 设变量a的值为0，b的值为-10，

输出示例为：c=-10
［提示：用条件运算符］
-----------------------------------------------------------------
#include <stdio.h>
int main ()

{
 int a,b,c;
	a= 0;
	b= -10;
	c= (a>b)?b:a;

printf ("c=%d",c);

return 0;
}
